## list git remote
git remote -v

## remove previous upstream
git remote remove upstream

## rename origin name
git remote rename origin upstream

## set upstream
git branch --set-upstream-to=origin/main main

## disable push upstream
git remote set-url --push upstream DISABLE

## add remote origin
git remote add origin git@github.com:mortexafarhadi/core.git

## change main branch
git branch -M main

## first push to origin
git push -u origin main

## add submodule to main git
git submodule add https://github.com/mortexafarhadi/core.git projects/core

## clone and pull submodules
git submodule update --init --recursive

## for fetch core (main)
git fetch upstream
git checkout main

git merge upstream/main   ## for clean history   -> git rebase upstream/main

## for fix conflict
git push origin main

##### for remove commited file from git history
## remove file from git cache
git rm --cached <filename>
git rm -r --cached <folder-name>   ## for folder
